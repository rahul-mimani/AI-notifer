Create a new file src/types.ts containing TypeScript type definitions for the domain of a change-impact analysis tool. Include these types and export all of them:
SurfaceSet — the complete public surface of one repo at one commit. Has provides: { endpoints: RestEndpoint[]; dtos: DtoType[]; beans: BeanDefinition[] } and consumes: { httpCalls: HttpCallSite[]; beanInjections: BeanInjection[] }.
RestEndpoint — { id: string; method: string; path: string; requestType: string | null; responseType: string; file: string; line: number; authAnnotation?: string } where id is ${method}:${normalizedPath}.
DtoType — { fqName: string; fields: DtoField[]; file: string; line: number }. DtoField is { name: string; type: string; nullable: boolean; required: boolean }.
BeanDefinition — { name: string; type: string; packageName: string; scope: string; qualifiers: string[]; conditionals: string[]; exported: boolean; publicMethods: MethodSignature[]; file: string; line: number }. MethodSignature is { name: string; returnType: string; parameters: { name: string; type: string }[] }.
HttpCallSite — { provider: string; method: string; path: string; responseType: string; usedFields: UsedField[]; file: string; line: number }. UsedField is { path: string; confidence: 'declared' | 'inferred' | 'unknown' }.
BeanInjection — { providerType: string; qualifier: string | null; calledMethods: CalledMethod[]; file: string; line: number; class: string; field: string }. CalledMethod is { signature: string; confidence: 'declared' | 'inferred' | 'unknown' }.
ChangeKind — a string union covering: 'ENDPOINT_ADDED', 'ENDPOINT_REMOVED', 'ENDPOINT_METHOD_CHANGED', 'DTO_FIELD_ADDED', 'DTO_FIELD_REMOVED', 'DTO_FIELD_RENAMED', 'DTO_FIELD_TYPE_CHANGED', 'DTO_FIELD_REQUIREDNESS_CHANGED', 'BEAN_ADDED', 'BEAN_REMOVED', 'BEAN_RENAMED', 'BEAN_METHOD_ADDED', 'BEAN_METHOD_REMOVED', 'BEAN_METHOD_SIGNATURE_CHANGED', 'BEAN_QUALIFIER_CHANGED', 'BEAN_CONDITIONAL_TIGHTENED'.
ChangeEvent — { id: string; kind: ChangeKind; surfaceKind: 'endpoint' | 'dto' | 'bean'; surfaceId: string; severity: 'breaking' | 'potentially_breaking' | 'additive'; before: unknown; after: unknown; riskFlags: string[] }.
StoreManifest — { schemaVersion: 1; repo: string; generatedAt: string; generatedFromCommit: string; provides: SurfaceSet['provides']; consumes: SurfaceSet['consumes'] }.
FreshnessInfo — { ageDays: number; commit: string; javaFilesChangedSince: number; flag: 'fresh' | 'stale' | 'significantly_drifted' }.
ImpactRecord — { recordId: string; consumer: string; changeEvent: ChangeEvent; coupling: 'http' | 'bean'; callSite: { file: string; line: number }; usedTarget: string; usedTargetConfidence: 'declared' | 'inferred' | 'unknown' | null; status: 'breaking' | 'review_recommended' | 'safe' | 'unknown'; reasoning: string[]; narrative?: string; migrationSuggestion?: string; manifestFreshness: FreshnessInfo }.
ConsumerImpact — { repo: string; manifest: { generatedAt: string; commit: string; ageDays: number; driftSinceManifest: { javaFilesChanged: number; flag: string } }; impacts: ImpactRecord[] }.
ImpactReport — { meta: { provider: string; base: string; head: string; generatedAt: string }; level1: { riskScore: number; changes: ChangeEvent[] }; level2: { consumersScanned: number; consumersUnreadable: number; consumers: ConsumerImpact[] } }.
Add JSDoc comments on top-level types. No implementation code. Types only.
